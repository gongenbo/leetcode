public boolean canPartition(int[] nums) {
    //求sum/2
    int sum = 0;
    for(int num: nums){
        sum += num;
    }
    if((sum & 1) ==1) return false;
    int target = sum/2;
    int len = nums.length;
    boolean[] dp =new boolean[target+1];
    dp[0] = true;
    //0,1背包
    for(int num: nums){
        for(int i = target;i>=num;i--){//从后往前，防止干扰
            dp[i] = dp[i]||dp[i-num];//dp[i][j] = dp[i-1][j] || dp[i-1][j-1] 不包括与包括;
        }
    }
    return dp[target];
}
